function Emoji() {
    return function(target: Object, key: string | symbol) {

        let val = target[key];

        const getter = () =>  {
            return val;
        };
        const setter = (next) => {
            console.log('updating flavor...');

        };

        Object.defineProperty(target, key, {

            set: setter,
            enumerable: true,
            configurable: true,
        });

    };
}
